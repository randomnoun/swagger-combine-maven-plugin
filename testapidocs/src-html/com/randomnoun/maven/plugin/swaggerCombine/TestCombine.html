021        // tests that $xrefs are resolved, and that $refs aren't
022        public void test1() throws IOException {
023                Log log = new SystemStreamLog();
024
025                SwaggerCombiner sc = new SwaggerCombiner();
026        sc.setRelativeDir(new File("src/test/resources/t1"));
027        sc.setFiles(new String[] { "input.yaml" });
028        sc.setLog(log);
029        
030                StringWriter w = new StringWriter();
031        sc.combine(w);
032
033        Path expectedPath = new File("src/test/resources/t1/expected-output.yaml").toPath();
034        List<String> expectedLines = Files.readAllLines(expectedPath);
035        
036        List<String> lines = Arrays.asList(w.toString().split("\n"));
037
038        Assert.assertEquals(expectedLines, lines);
039        }
