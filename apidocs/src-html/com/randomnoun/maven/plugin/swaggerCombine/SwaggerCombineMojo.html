102    // private method in FileSetManager
103
104        private DirectoryScanner scan(FileSet fileSet) {
105                File basedir = new File(fileSet.getDirectory());
106                if (!basedir.exists() || !basedir.isDirectory()) {
107                        return null;
108                }
109
110                DirectoryScanner scanner = new DirectoryScanner();
111
112                String[] includesArray = fileSet.getIncludesArray();
113                String[] excludesArray = fileSet.getExcludesArray();
114
115                if (includesArray.length > 0) {
116                        scanner.setIncludes(includesArray);
117                }
118
119                if (excludesArray.length > 0) {
120                        scanner.setExcludes(excludesArray);
121                }
122
123                if (fileSet.isUseDefaultExcludes()) {
124                        scanner.addDefaultExcludes();
125                }
126
127                scanner.setBasedir(basedir);
128                scanner.setFollowSymlinks(fileSet.isFollowSymlinks());
129
130                scanner.scan();
131
132                return scanner;
133        }

138    /**
139     * Generates the JAR.
140     * 
141     * @return The instance of File for the created archive file.
142     * @throws MojoExecutionException in case of an error.
143     */
144        public File createCombinedSwaggerYaml() throws MojoExecutionException {
145                if (outputDirectory == null) {
146                        throw new IllegalArgumentException("Missing outputDirectory");
147                }
148                if (finalName == null) {
149                        throw new IllegalArgumentException("Missing finalName");
150                }
151                File destFile = new File(outputDirectory, finalName);
152
153                // filesets aren't order-preserving.
154                // which is kind of annoying. could sort the Files just to get some
155                // deterministic order I suppose
156
157                if (destFile.exists() && destFile.isDirectory()) {
158                        throw new MojoExecutionException(destFile + " is a directory.");
159                }
160                if (destFile.exists() && !destFile.canWrite()) {
161                        throw new MojoExecutionException(destFile + " is not writable.");
162                }
163
164                getLog().info("Creating combined swagger file " + destFile.getAbsolutePath());
165
166                if (!destFile.getParentFile().exists()) {
167                        // create the parent directory...
168                        if (!destFile.getParentFile().mkdirs()) {
169                                // Failure, unable to create specified directory for some unknown reason.
170                                throw new MojoExecutionException("Unable to create directory or parent directory of " + destFile);
171                        }
172                }
173                if (destFile.exists()) {
174                        // delete existing
175                        if (!destFile.delete()) {
176                                throw new MojoExecutionException("Unable to delete existing file " + destFile);
177                        }
178                }
179
180                DirectoryScanner scanner = scan(fileset);
181                String[] files = scanner.getIncludedFiles(); // also performs exclusion. So way to go, maven.
182
183                // let's just have a single output file then.
184                try {
185                        SwaggerCombiner sc = new SwaggerCombiner();
186                        sc.setRelativeDir(new File(fileset.getDirectory()));
187                        sc.setFiles(files);
188                        sc.setLog(getLog());
189                        
190                        FileOutputStream fos = new FileOutputStream(destFile);
191                        if (filtering) {
192                                MavenResourcesExecution mre = new MavenResourcesExecution();
193                        mre.setMavenProject( project );
194                        mre.setFileFilters( null );
195                        mre.setEscapeWindowsPaths( true );
196                        mre.setMavenSession( session );
197                        mre.setInjectProjectBuildFilters( true );
198                        
199                        // there's probably a better way of getting a plexus Logger instance
200                        // but for the life of me I can't work it out 
201                        // after about 15 levels of indirection they end up not-properly wrapping slf4j anyway.
202                        ConsoleLoggerManager clm = new ConsoleLoggerManager();
203                        Logger logger = clm.getLoggerForComponent("SwaggerCombineMojo");
204                        
205                        DefaultMavenFileFilter dmff = new DefaultMavenFileFilter();
206                        dmff.enableLogging(logger);
207                        getLog().info("logger is " + logger);
208                                try {
209                                        List<FileUtils.FilterWrapper> filterWrappers = dmff.getDefaultFilterWrappers( mre );
210                                        sc.setFilterWrappers(filterWrappers);
211                                } catch (MavenFilteringException e) {
212                                        throw new IllegalStateException("Coult not get default filter wrappers", e);
213                                }
214                        }
215                        
216                        PrintWriter w = new PrintWriter(fos);
217                        sc.combine(w);
218                        fos.close();
219                } catch (IOException ioe) {
220                        // trouble at the mill
221                        throw new MojoExecutionException("Could not create combined swagger file", ioe); 
222                }
223                return destFile;
224
225        }
